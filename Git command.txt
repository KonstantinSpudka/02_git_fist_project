git init - инициализируем репозиторий один раз (создаем в корне .git) и он следит за изменениями

git status - проверяем статус
git add . - добавление в индексированую область все(станет зеленым все)

git commit - создает комит через редактор
git commit -m "" - создает комит без редактор
--amend - изменить/отменить последний комит

git log - просмотр всех коментов полностью
git log --oneline - просмотр всех коментов максимально коротко

git remote -v - перечень всех удаленных репозитов
git remote add origin htpps://path.git - добавление удаленного репозита
git push origin main - отправляем на удаленный с локального
git clone https://path.git - копирует с удаленки на локальный
